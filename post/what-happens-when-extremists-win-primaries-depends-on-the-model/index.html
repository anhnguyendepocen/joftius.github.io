margins <- c(.05, .1, .15, .2, .3)
rad_dem_penalty <- data.frame(sapply(margins, function(m) {
  data = subset(data, margin < m)
  model <- lm(dv ~ treat*dem + treat * rv, data)
  return(c(nrow(data), coef(model)["treat"] + coef(model)["treat:dem"]))
}))
names(rad_dem_penalty) <- margins
rownames(rad_dem_penalty) <- c("Sample size", "Effect size")
kable(t(rad_dem_penalty), digits = 3)
